// GET: /VisitsReport/        public ActionResult Index()        {            //List<Resident> Residents = db.Residents;            DateTime Start = DateTime.Now.AddMonths(-1);            DateTime End = DateTime.Now;            var visits = db.Visits                .Where(v => v.VisitTime > 16 && v.CheckIn >= Start && v.CheckIn <= End)                .Where(v => v.Resident.Visits.Count() > 4 || v.Visitor.Visits.Count() > 4)                .Include(v => v.Resident);            VisitsReportViewModel report = new VisitsReportViewModel();            return View(report);            //return View();        }
